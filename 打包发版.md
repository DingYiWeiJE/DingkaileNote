### 依赖重定向配置

```js
const { getDefaultConfig } = require("@expo/metro-config");
const { withNativeWind } = require("nativewind/metro");
const path = require('node:path');
const { IS_DEV } = require('./env.config');

const config = getDefaultConfig(__dirname);

console.log('🔍 Metro 环境配置:', {
  IS_DEV,
  mode: IS_DEV ? '本地开发' : '生产构建'
});

if (IS_DEV) {
  // 本地开发：使用 pythra-sdk 热更新
  const pythraPath = path.resolve(__dirname, '../pythra-sdk');
  config.watchFolders = [pythraPath];

  config.resolver.extraNodeModules = {
    'pythra-sdk': pythraPath,
    'react': path.resolve(__dirname, 'node_modules/react'),
    'react-native': path.resolve(__dirname, 'node_modules/react-native'),
  };

  console.log('🔥 本地开发模式：使用 pythra-sdk 热更新');
} else {
  // EAS Build：使用内置的 lib/pythra-sdk
  config.resolver.extraNodeModules = {
    'pythra-sdk': path.resolve(__dirname, 'lib/pythra-sdk'),
  };

  console.log('📦 构建模式：使用内置 lib/pythra-sdk');
}

// 配置解析器
config.resolver.nodeModulesPaths = [
  path.resolve(__dirname, 'node_modules'),
];

const ALIASES = {
  tslib: path.resolve(__dirname, "node_modules/tslib/tslib.es6.js"),
};

config.resolver.resolveRequest = (context, moduleName, platform) => {
  // 在本地开发时，将 @/lib/pythra-sdk 重定向到外部 pythra-sdk
  if (IS_DEV && moduleName === '@/lib/pythra-sdk') {
    return context.resolveRequest(context, 'pythra-sdk', platform);
  }

  return context.resolveRequest(
    context,
    ALIASES[moduleName] ?? moduleName,
    platform
  );
};

// 使用 NativeWind 配置
module.exports = withNativeWind(config, { input: "./global.css" });

```

## **步骤 1：配置防火墙**

首先，检查防火墙是否允许外部访问 **端口 8000**：

```
bash


复制编辑
sudo ufw status
```

如果防火墙已启用，并且没有允许 8000 端口的规则，你需要手动开放：

```
bash


复制编辑
sudo ufw allow 8000/tcp
sudo ufw reload
```

如果防火墙未启用，建议先启用：

```
bash


复制编辑
sudo ufw enable
```

然后再次确认规则：

```
bash


复制编辑
sudo ufw status
```

------

## **步骤 2：配置 Nginx 作为反向代理**

虽然 Uvicorn 直接监听 8000 端口可以被访问，但通常我们会使用 **Nginx 作为反向代理**，让它代理请求到 Uvicorn，以提高安全性和性能。

### **1. 创建 Nginx 配置文件**

新建 Nginx 配置：

```
bash


复制编辑
sudo nano /etc/nginx/sites-available/uvicorn
```

添加以下内容：

```
nginx


复制编辑
server {
    listen 80;
    server_name your-server-ip; # 或者你的域名

    location / {
        proxy_pass http://127.0.0.1:8000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
}
```

**注意**：

- **`server_name your-server-ip;`** 这里要改成你的 **公网 IP** 或域名。
- **Uvicorn 运行在 `127.0.0.1:8000`**，所以 **Nginx 只允许本机访问**，外部用户通过 **Nginx 访问端口 80**。

------

### **2. 启用配置**

创建符号链接，使该配置生效：

```
bash


复制编辑
sudo ln -s /etc/nginx/sites-available/uvicorn /etc/nginx/sites-enabled/
```

测试 Nginx 配置是否正确：

```
bash


复制编辑
sudo nginx -t
```

如果没有错误，重新加载 Nginx：

```
bash


复制编辑
sudo systemctl restart nginx
```

------

## **步骤 3：让 Uvicorn 在后台运行**

你当前使用的 `ps aux | grep uvicorn` 显示它是手动启动的，这样如果服务器重启，Uvicorn 也会停止。推荐使用 **systemd** 让它作为服务运行：

```
bash


复制编辑
sudo nano /etc/systemd/system/uvicorn.service
```

添加以下内容：

```
ini


复制编辑
[Unit]
Description=Uvicorn FastAPI Server
After=network.target

[Service]
User=ubuntu
Group=ubuntu
WorkingDirectory=/home/ubuntu
ExecStart=/usr/bin/python3 /home/ubuntu/.local/bin/uvicorn main:app --host 127.0.0.1 --port 8000
Restart=always

[Install]
WantedBy=multi-user.target
```

**启用并启动 Uvicorn：**

```
bash


复制编辑
sudo systemctl daemon-reload
sudo systemctl enable uvicorn
sudo systemctl start uvicorn
```

检查状态：

```
bash


复制编辑
sudo systemctl status uvicorn
```

------

## **步骤 4：测试外部访问**

现在，你可以通过浏览器访问：

```
bash


复制编辑
http://your-server-ip
```

如果你看到 **FastAPI 文档** 或你的 API 正常工作，说明 Nginx 已经成功代理到 Uvicorn。

------

## **可选：配置 HTTPS**

为了让网站更加安全，可以使用 **Let's Encrypt** 免费 SSL 证书：

```
bash


复制编辑
sudo apt install certbot python3-certbot-nginx
sudo certbot --nginx -d yourdomain.com
```

然后，自动更新证书：

```
bash


复制编辑
sudo crontab -e
```

添加：

```
css


复制编辑
0 3 * * * certbot renew --quiet
```

这会每天凌晨 3 点自动更新证书。

------

## **总结**

- **开放防火墙端口**
- **使用 Nginx 作为反向代理**
- **使用 systemd 让 Uvicorn 在后台运行**
- **（可选）配置 HTTPS**
