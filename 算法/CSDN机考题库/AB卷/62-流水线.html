import java.util.ArrayList;
import java.util.Arrays;
import java.util.Collections;
import java.util.List;
import java.util.Scanner;


public class Main {
	public static void sovle(String[] strs, String[] jobs) {
		int medValue = Integer.parseInt(strs[0]);
		int[] array = new int[jobs.length];
		for (int j = 0; j < jobs.length; j++) {
			array[j] = Integer.parseInt(jobs[j]);
		}
		// æ’åº
		Arrays.sort(array);
		List<MedicBean> myList = new ArrayList<>();
		for (int j = 0; j < array.length; ++j) {
			if (myList.size() < medValue) {
				myList.add(new MedicBean(array[j], array[j]));
			} else {
				// æ‰¾åˆ°æ•°ç»„ä¸­ totalæœ€å°çš„medic åŠ è¿›å»
				Collections.sort(myList);
				MedicBean medic = myList.get(0);
				medic.sum += array[j];
			}
		}
		// æ’åº
		Collections.sort(myList);
		System.out.println(myList.get(myList.size() - 1).sum);
	}

	static class MedicBean implements Comparable {
		public int end;
		public int sum;

		public MedicBean(int end, int sum) {
			this.end = end;
			this.sum = sum;
		}

		@Override
		public int compareTo(Object obj) {
			MedicBean medic = (MedicBean) obj;
			return this.sum - medic.sum;
		}
	}

	public static void main(String[] args) {
		try {
			Scanner s = new Scanner(System.in);
			String[] strs = s.nextLine().split(" ");
			String[] jobs = s.nextLine().split(" ");
			sovle(strs, jobs);
		} catch (Exception e) {
			e.printStackTrace();
			System.out.println("input error");
		}
	}
}
