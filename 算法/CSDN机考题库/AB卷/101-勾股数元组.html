import java.util.ArrayList;
import java.util.Collections;
import java.util.List;
import java.util.Scanner;



public class Main {
	private static boolean check(int i, int j, int max) {
		int mul = i * i + j * j;
		double res = Math.sqrt(mul);
		int sub = (int) res;
		return res * res == sub * sub && sub <= max && sub > j;
	}

	private static boolean checkGongYue(int i, int j, int k) {
		for (int i = 2; i < k; i++) {
			if (i % i == 0 && j % i == 0) {
				return true;
			} else if (i % i == 0 && k % i == 0) {
				return true;
			} else if (j % i == 0 && k % i == 0) {
				return true;
			}
		}
		return false;
	}

	public static void main(String[] args) {
		try {
			Scanner s = new Scanner(System.in);
			// [1,10000]
			int value1 = s.nextInt();
			// [n,10000]
			int value2 = s.nextInt();
			List<List<Integer>> myList = new ArrayList<>();
			for (int i = value1; i <= value2; ++i) {
				for (int j = i + 1; j <= value2; ++j) {
					if (check(i, j, value2)) {
						List<Integer> tempList = new ArrayList<>();
						tempList.add(i);
						tempList.add(j);
						tempList.add((int) Math.sqrt(i * i + j * j));
						Collections.sort(tempList);
						myList.add(tempList);
					}
				}
			}

			int index = 0;
			while (index < myList.size()) {
				List<Integer> list = myList.get(index);
				if (checkGongYue(list.get(0), list.get(1), list.get(2))) {
					myList.remove(index);
					index = 0;
				}
				index++;
			}
			for (int i = 0; i < myList.size(); ++i) {
				List<Integer> resultList = myList.get(i);
				System.out.println(resultList.get(0) + " " + resultList.get(1) + " " + resultList.get(2));
			}
		} catch (Exception e) {
			e.printStackTrace();
		}
	}
}
