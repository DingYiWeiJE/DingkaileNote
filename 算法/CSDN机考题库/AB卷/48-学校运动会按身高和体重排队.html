import java.util.Arrays;
import java.util.Scanner;

public class Main {
  public static void main(String[] args) {
    try (Scanner s = new Scanner(System.in)) {
      int n = Integer.parseInt(s.nextLine());
      String heights = s.nextLine();
      String weights = s.nextLine();
      solveMethod(n, heights, weights);
    }
  }

  private static void solveMethod(int n, String heights, String weights) {
    String[] heightsArr = heights.split(" ");
    String[] weightsArr = weights.split(" ");
    int[][] ints = new int[n][3];
    for (int i = 0; i < n; ++i) {
      ints[i][0] = i + 1;
      ints[i][1] = Integer.parseInt(heightsArr[i]);
      ints[i][2] = Integer.parseInt(weightsArr[i]);
    }
    Arrays.sort(ints, (stu1, stu2) -> stu1[1] == stu2[1] ? stu1[2] - stu2[2] : stu1[1] - stu2[1]);

    for (int i = 0; i < n; ++i) {
      System.out.print(ints[i][0]);
      if (i != n - 1) {
        System.out.print(" ");
      }
    }
  }
}
