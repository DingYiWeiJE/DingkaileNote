import java.util.Arrays;
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        try (Scanner s = new Scanner(System.in)) {
            int n = Integer.parseInt(s.nextLine());
            String[] lines = new String[n];
            for (int i = 0; i < n; ++i) {
                lines[i] = s.nextLine();
            }
            solveMethod(n, lines);
        }
    }

    private static void solveMethod(int n, String[] lines) {
        int res = 0;
        int[][] nums = new int[n][];
        for (int i = 0; i < n; ++i) {
            nums[i] = Arrays.stream(lines[i].split(","))
                      .mapToInt(Integer::parseInt)
                      .toArray();
        }
        for (int i = 0; i < n; ++i) {
            int max = Integer.MIN_VALUE;
            for (int j = 0; j < nums[i].length; ++j) {
                int[] shifted = new int[nums[i].length];
                System.arraycopy(nums[i], j, shifted, 0, nums[i].length - j);
                System.arraycopy(nums[i], 0, shifted, nums[i].length - j, j);
                String binInt = Arrays.toString(shifted)
                                .replaceAll("\\W+", "");
                int sum = Integer.parseInt(binInt, 2);
                if (sum > max) {
                    max = sum;
                }
            }
            res += max;
        }
        System.out.print(res);
    }
}
