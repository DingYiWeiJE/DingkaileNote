import java.util.ArrayList;
import java.util.Collections;
import java.util.List;
import java.util.Scanner;

public class Main {
	private static void solve(String[] nums) {
		int stepOneMinValue = 2;
		int stepOneMaxValue = nums.length / 2;
		int minStepValue = Integer.MAX_VALUE;
		int current = 0;
		int stepTmp = 0;
		for (int i = stepOneMinValue; i < stepOneMaxValue; i++) {
			int targetValue = nums.length - 1;
			current = i;
			++stepTmp;
			while (current < targetValue) {
				current += Integer.parseInt(nums[current]);
				++stepTmp;
				if (targetValue == current) {
					minStepValue = Math.min(minStepValue, stepTmp);
					break;
				}
				if (current > targetValue) {
					stepTmp = 0;
					current = 0;
					break;
				}
			}
		}
		if (minStepValue <= nums.length) {
			System.out.println(minStepValue);
		} else {
			System.out.println(-1);
		}

	}

	public static void main(String[] args) {
		try {
			Scanner scan = new Scanner(System.in);
			// æœ€å¤§ä¸º100ä¸ªæˆå‘˜

				return;
			}
			solve(nums);
		} catch (Exception e) {
			e.printStackTrace();
			System.out.println("input error");
		}
	}
}
