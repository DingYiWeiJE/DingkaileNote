import java.util.*;

public class Main {
    private static int res = 0;
    private static int target;
    private static Set<List<Integer>> numsSet = new HashSet<>();

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int[] nums = Arrays.stream(sc.nextLine().split(" ")).mapToInt(Integer::parseInt).toArray();
        int k = sc.nextInt();
        target = sc.nextInt();

        Arrays.sort(nums);
        combine(nums, k, new ArrayList<>(), 0, 0);

        System.out.println(res);
    }

    public static void combine(int[] nums, int n, List<Integer> list, int index, int sum) {
        if (n == 0) {
            if (sum == target) {
                numsSet.add(new ArrayList<>(list));
                res++;
            }
        } else {
            for (int i = index; i < nums.length; i++) {
                if (i > index && nums[i] == nums[i - 1]) {
                    continue;
                }
                if (sum + nums[i] > target) {
                    break;
                }
                list.add(nums[i]);
                combine(nums, n - 1, list, i + 1, sum + nums[i]);
                list.remove(list.size() - 1);
            }
        }
    }
}
