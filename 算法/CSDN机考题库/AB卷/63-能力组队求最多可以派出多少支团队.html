import java.util.*;
import java.util.stream.IntStream;

public class Main {

    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);

        int n = sc.nextInt();
        List<Integer> list = new ArrayList<>();

        for (int i = 0; i < n; i++) {
            list.add(sc.nextInt());
        }

        int m = sc.nextInt();

        long count = list.stream().filter(i -> i >= m).count();

        count += IntStream.range(0, list.size())
                 .filter(i -> list.get(i) < m)
                 .filter(i -> list.indexOf(m - list.get(i)) > i)
                 .count();

        System.out.println(count);
    }

}
