import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        String user_input = scanner.nextLine();
        int ret = solveMethod(user_input);
        System.out.println(ret);
    }


    private static int solveMethod(String quack) {
        String base = "quack";
        int[] states = new int[base.length()];
        int counts = 0;

        for (char c : quack.toCharArray()) {
            if (c == base.charAt(0)) {
                states[0]++;
            } else {
                int index = base.indexOf(c);
                states[index - 1]--;
                states[index]++;
                if (c == base.charAt(base.length() - 1)) {
                    states[states.length - 1]--;
                }
            }

            counts = Math.max(counts, sum(states));
            if (containsNegative(states)) {
                return -1;
            }
        }

        if (sum(states) != 0) {
            return -1;
        }
        return counts;
    }

    private static int sum(int[] array) {
        int sum = 0;
        for (int num : array) {
            sum += num;
        }
        return sum;
    }

    private static boolean containsNegative(int[] array) {
        for (int num : array) {
            if (num < 0) {
                return true;
            }
        }
        return false;
    }
}
