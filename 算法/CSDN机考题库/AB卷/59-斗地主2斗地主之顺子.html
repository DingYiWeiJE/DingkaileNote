import java.util.*;

public class Main {
    static Map<String, Integer> mp = new HashMap<>();

    static void init() {
        mp.put("2", 13);
        mp.put("3", 0);
        mp.put("4", 1);
        mp.put("5", 2);
        mp.put("6", 3);
        mp.put("7", 4);
        mp.put("8", 5);
        mp.put("9", 6);
        mp.put("10", 7);
        mp.put("J", 8);
        mp.put("Q", 9);
        mp.put("K", 10);
        mp.put("A", 11);
    }

    static void print(int x, int y) {
        char[] card = {'3', '4', '5', '6', '7', '8', '9', '1', 'J', 'Q', 'K', 'A'};
        for (int i = x; i < y; ++i) {
            if (i != x) {
                System.out.print(" ");
            }
            System.out.print(card[i]);
        }
        System.out.println();
    }

    public static void main(String[] args) {
        init();
        Scanner scanner = new Scanner(System.in);
        while (scanner.hasNextLine()) {
            String str = scanner.nextLine();
            int[] num = new int[15];
            Arrays.fill(num, 0);
            String[] s = str.split(" ");
            for (String value : s) {
                num[mp.get(value)]++;
            }

            boolean ans = false;
            for (int i = 0; i < 12; ++i) {
                int j = i;
                while (j < 12) {
                    if (num[j] == 0) {
                        break;
                    }
                    ++j;
                }
                if (j > i + 4) {
                    ans = true;
                    for (int k = i; k < j; ++k) {
                        num[k]--;
                    }
                    print(i, j);
                }
            }
            if (!ans) {
                System.out.println("NO");
            }
        }
        scanner.close();
    }
}
